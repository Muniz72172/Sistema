    function gerarLista() {
      const data = document.getElementById("data").value;
      const destino = document.getElementById("destino").value;
      if (!data || !destino) {
        alert("Informe data e destino.");
        return;
      }
      google.script.run.withSuccessHandler(function(lista) {
        const tabela = document.getElementById("tabelaPresenca");
        tabela.innerHTML = "";
        lista.forEach((p, i) => {
          const tr = document.createElement("tr");
          tr.innerHTML = `
            <td>${i + 1}</td>
            <td>${p.nip}</td>
            <td>${p.nome}</td>
            <td>${p.assento}</td>
            <td>${p.pcd ? "Sim" : "Não"}</td>
            <td>${p.acompanhante ? "Sim" : "Não"}</td>
            <td></td>`;
          tabela.appendChild(tr);
        });
        google.script.run.registrarEventoOnibus(data, destino, "Lista de Presença Gerada");
      }).gerarListaPresenca(data, destino);
    }
  </script>
