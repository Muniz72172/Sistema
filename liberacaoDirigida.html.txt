    function liberar() {
      const data = document.getElementById("data").value;
      const destino = document.getElementById("destino").value;
      const assento = document.getElementById("assento").value;
      const nip = document.getElementById("nip").value.trim();
      const justificativa = document.getElementById("justificativa").value.trim();

      if (!data || !destino || !assento || !nip || !justificativa) {
        alert("Preencha todos os campos.");
        return;
      }

      google.script.run.withSuccessHandler(function() {
        document.getElementById("mensagem").innerText = "Assento liberado com sucesso.";
      }).liberarAssentoParaUsuario(data, destino, assento, nip, justificativa);
    }
